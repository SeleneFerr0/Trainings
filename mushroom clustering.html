import time
from sklearn.metrics import silhouette_score
from sklearn.cluster import KMeans

def cluster(X):
    '''
    K-Means
    
    
        X
    
        best_clusters
        score
        
    '''
    score = -1
    duration = 0
    start = time.time()
    # K215
    for clusters_num in range(2, 15):  
        # TODO
        # kmeansclusters_num
        kmeans =  KMeans(n_clusters=clusters_num)
        
        # TODO
        # kmeansXXtest_preds
        kmeans.fit(X)
        test_preds = kmeans.predict(X)

        # TODO
        # silhouette_scoretest_score
        test_score = silhouette_score(X, test_preds)
        
        # 
        print("component score is ", test_score)

        #
        if (score < test_score):
            best_clustersnum = clusters_num
            score = test_score
            best_clusters=kmeans

    # 
    end = time.time()      
    duration += end - start

    # 
    print(': {:.4f}s'.format(duration))
    print('k{}{:.3f}'.format(best_clustersnum, score))
    
    return best_clusters, score
